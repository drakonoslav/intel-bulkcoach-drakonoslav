Below is a developer-ready BulkCoach build packet v1 for the Vitals Dashboard + Vitals Daily Recommendation Engine.

It includes:
	•	TypeScript interfaces
	•	enums
	•	scoring functions
	•	SQL table definitions
	•	API response shapes
	•	implementation notes

I’m keeping naming practical and app-oriented so it can drop into a React Native + Node/Express + Postgres stack without drama.

5. Utility functions

export function clamp(value: number, min: number, max: number): number {
  return Math.max(min, Math.min(max, value));
}

export function round(value: number, digits = 0): number {
  const factor = Math.pow(10, digits);
  return Math.round(value * factor) / factor;
}

export function safeRatio(a: number | null, b: number | null): number | null {
  if (a == null || b == null || b === 0) return null;
  return a / b;
}

export function avg(values: Array<number | null | undefined>): number | null {
  const valid = values.filter((v): v is number => typeof v === "number" && !Number.isNaN(v));
  if (!valid.length) return null;
  return valid.reduce((s, v) => s + v, 0) / valid.length;
}

export function scoreBand(
  value: number | null,
  bands: Array<{ min?: number; max?: number; score: number }>
): number {
  if (value == null) return 0;
  for (const band of bands) {
    const minOk = band.min == null || value >= band.min;
    const maxOk = band.max == null || value <= band.max;
    if (minOk && maxOk) return band.score;
  }
  return 0;
}

export function normalizeFivePoint(value: number | null): number | null {
  if (value == null) return null;
  return clamp(((value - 1) / 4) * 10, 0, 10);
}

export function normalizeThreePointZeroBased(value: number | null): number | null {
  if (value == null) return null;
  return clamp((value / 3) * 10, 0, 10);
}

export function reverseFivePoint(value: number | null): number | null {
  if (value == null) return null;
  return clamp(((5 - value) / 4) * 10, 0, 10);
}

6. Acute score scoring functions

export interface AcuteScoreContext {
  today: DailyVitalsRecord;
  refs: RollingReferences;
  baselines: UserVitalsBaselines;
  yesterdayLiftStrainScore: number | null;
  yesterdayCardioMode: CardioMode | null;
}

export function scoreHrvState(ctx: AcuteScoreContext): ScoreBreakdownItem {
  const hrvToday = ctx.today.heart.hrvMs;
  const hrv7 = ctx.refs.hrv7dAvg;
  const hrvYear = ctx.baselines.hrvYearAvg;

  const ratio = safeRatio(hrvToday, hrv7);
  const yearRatio = safeRatio(hrvToday, hrvYear);

  let score = 0;
  if (ratio == null) score = 0;
  else if (ratio >= 1.10) score = 22;
  else if (ratio >= 1.03) score = 19;
  else if (ratio >= 0.97) score = 15;
  else if (ratio >= 0.90) score = 9;
  else score = 4;

  if (yearRatio != null) {
    if (yearRatio > 1.15) score += 1;
    if (yearRatio < 0.85) score -= 1;
  }

  return {
    key: "hrv_state",
    label: "HRV State",
    score: clamp(score, 0, 22),
    maxScore: 22,
    note: ratio == null ? "Missing HRV" : `HRV ratio vs 7d avg: ${round(ratio, 2)}`,
  };
}

export function scoreRhrState(ctx: AcuteScoreContext): ScoreBreakdownItem {
  const rhrToday = ctx.today.heart.restingHrBpm;
  const rhr7 = ctx.refs.rhr7dAvg;
  const rhrYear = ctx.baselines.rhrYearAvg;

  let score = 0;
  if (rhrToday == null || rhr7 == null) {
    score = 0;
  } else {
    const delta = rhrToday - rhr7;
    if (delta <= -3) score = 18;
    else if (delta <= -1) score = 16;
    else if (delta <= 1) score = 13;
    else if (delta <= 3) score = 8;
    else if (delta <= 5) score = 4;
    else score = 1;

    if (rhrYear != null) {
      if (rhrToday <= rhrYear - 3) score += 1;
      if (rhrToday >= rhrYear + 5) score -= 1;
    }
  }

  return {
    key: "rhr_state",
    label: "RHR State",
    score: clamp(score, 0, 18),
    maxScore: 18,
    note:
      rhrToday == null || rhr7 == null
        ? "Missing RHR"
        : `RHR delta vs 7d avg: ${round(rhrToday - rhr7, 1)} bpm`,
  };
}

export function scoreSleepQuantity(ctx: AcuteScoreContext): ScoreBreakdownItem {
  const sleep = ctx.today.sleep.sleepDurationMin;
  let score = 0;

  if (sleep == null) score = 0;
  else if (sleep >= 450 && sleep <= 510) score = 15;
  else if ((sleep >= 420 && sleep <= 449) || (sleep >= 511 && sleep <= 540)) score = 12;
  else if (sleep >= 390 && sleep <= 419) score = 8;
  else if (sleep >= 360 && sleep <= 389) score = 4;
  else score = 1;

  const eff = ctx.today.sleep.sleepEfficiencyPct;
  if (eff != null && eff >= 90) score += 1;

  return {
    key: "sleep_quantity",
    label: "Sleep Quantity",
    score: clamp(score, 0, 15),
    maxScore: 15,
    note: sleep == null ? "Missing sleep duration" : `${sleep} min sleep`,
  };
}

export function scoreSleepRegularity(ctx: AcuteScoreContext): ScoreBreakdownItem {
  const midpoint = ctx.today.sleep.sleepMidpointMin;
  const midpoint7 = ctx.refs.sleepMidpoint7dAvg;

  let score = 0;
  let note = "Missing midpoint data";

  if (midpoint != null && midpoint7 != null) {
    const shift = Math.abs(midpoint - midpoint7);
    note = `Midpoint shift: ${round(shift, 0)} min`;

    if (shift <= 20) score = 8;
    else if (shift <= 40) score = 6;
    else if (shift <= 60) score = 4;
    else if (shift <= 90) score = 2;
    else score = 0;
  }

  return {
    key: "sleep_regularity",
    label: "Sleep Regularity",
    score,
    maxScore: 8,
    note,
  };
}

export function scoreBodyweightStability(ctx: AcuteScoreContext): ScoreBreakdownItem {
  const weight = ctx.today.bodyComp.bodyWeightLb;
  const weight7 = ctx.refs.bodyWeight7dAvg;
  let score = 0;
  let note = "Missing body weight";

  if (weight != null && weight7 != null && weight7 > 0) {
    const deltaPct = ((weight - weight7) / weight7) * 100;
    const absPct = Math.abs(deltaPct);
    note = `Weight delta vs 7d avg: ${round(deltaPct, 2)}%`;

    if (absPct <= 0.4) score = 5;
    else if (absPct <= 0.8) score = 4;
    else if (absPct <= 1.2) score = 2;
    else score = 0;
  }

  return {
    key: "bodyweight_stability",
    label: "Bodyweight Stability",
    score,
    maxScore: 5,
    note,
  };
}

export function scoreSubjectiveDrive(ctx: AcuteScoreContext): ScoreBreakdownItem {
  const values = [
    normalizeFivePoint(ctx.today.subjective.libidoScore),
    normalizeThreePointZeroBased(ctx.today.subjective.morningErectionScore),
    normalizeFivePoint(ctx.today.subjective.motivationScore),
    normalizeFivePoint(ctx.today.subjective.mentalDriveScore),
  ];

  const value = avg(values);
  return {
    key: "subjective_drive",
    label: "Subjective Drive",
    score: clamp(round(value ?? 0, 0), 0, 10),
    maxScore: 10,
    note: value == null ? "Missing subjective entries" : `Average subjective drive ${round(value, 1)}/10`,
  };
}

export function scoreJointSoreness(ctx: AcuteScoreContext): ScoreBreakdownItem {
  const values = [
    reverseFivePoint(ctx.today.subjective.sorenessScore),
    reverseFivePoint(ctx.today.subjective.jointFrictionScore),
  ];
  const value = avg(values);

  return {
    key: "joint_soreness",
    label: "Joint / Soreness State",
    score: clamp(round(value ?? 0, 0), 0, 10),
    maxScore: 10,
    note: value == null ? "Missing soreness/joint data" : `Recovery comfort ${round(value, 1)}/10`,
  };
}

export function scoreYesterdayLiftStrain(ctx: AcuteScoreContext): ScoreBreakdownItem {
  const strain = ctx.yesterdayLiftStrainScore;
  let score = 0;

  if (strain == null) score = 0;
  else if (strain <= 35) score = 7;
  else if (strain <= 55) score = 5;
  else if (strain <= 70) score = 3;
  else score = 1;

  return {
    key: "yesterday_lift_strain",
    label: "Yesterday Lift Strain",
    score,
    maxScore: 7,
    note: strain == null ? "Missing yesterday lift strain" : `Yesterday lift strain ${strain}`,
  };
}

export function scoreYesterdayCardioStrain(ctx: AcuteScoreContext): ScoreBreakdownItem {
  let score = 0;

  switch (ctx.yesterdayCardioMode) {
    case CardioMode.RECOVERY_WALK:
      score = 5;
      break;
    case CardioMode.ZONE_2:
      score = 4;
      break;
    case CardioMode.ZONE_3:
      score = 2;
      break;
    default:
      score = 0;
  }

  return {
    key: "yesterday_cardio_strain",
    label: "Yesterday Cardio Strain",
    score,
    maxScore: 5,
    note: ctx.yesterdayCardioMode ?? "Missing yesterday cardio mode",
  };
}

export function calculateAcuteScore(ctx: AcuteScoreContext): AcuteScoreResult {
  const breakdown = [
    scoreHrvState(ctx),
    scoreRhrState(ctx),
    scoreSleepQuantity(ctx),
    scoreSleepRegularity(ctx),
    scoreBodyweightStability(ctx),
    scoreSubjectiveDrive(ctx),
    scoreJointSoreness(ctx),
    scoreYesterdayLiftStrain(ctx),
    scoreYesterdayCardioStrain(ctx),
  ];

  const score = clamp(
    breakdown.reduce((sum, item) => sum + item.score, 0),
    0,
    100
  );

  return { score, breakdown };
}

export function scoreCardioMonotony(ctx: ResourceScoreContext): ScoreBreakdownItem {
  const z2 = ctx.refs.cardioZone2Count7d;
  const z3 = ctx.refs.cardioZone3Count7d;
  const easy = ctx.refs.cardioRecoveryCount7d;

  const identicalMax = Math.max(z2, z3, easy);

  if ((z2 === 3 && z3 === 3 && easy === 1) || identicalMax <= 3) score = 10;
  else if (identicalMax <= 4) score = 7;
  else if (identicalMax <= 5) score = 4;
  else score = 1;

  return {
    key: "cardio_monotony",
    label: "Cardio Distribution 7d",
    score,
    maxScore: 10,
    note: `Z2:${z2} Z3:${z3} Easy:${easy}`,
  };
}

export function calculateResourceScore(ctx: ResourceScoreContext): ResourceScoreResult {
  const breakdown = [
    scoreCalorieAdherence7d(ctx),
    scoreProteinAdequacy7d(ctx),
    scoreFatFloorAdherence7d(ctx),
    scoreCarbAdequacyAroundTraining(ctx),
    scoreWeightTrend(ctx),
    scoreWaistTrend(ctx),
    scoreFfmTrend(ctx),
    scoreStrengthTrend(ctx),
    scoreCardioMonotony(ctx),
  ];

  const score = clamp(
    breakdown.reduce((sum, item) => sum + item.score, 0),
    0,
    100
  );

  return { score, breakdown };
}

export function scoreHrv28dTrend(ctx: SeasonalScoreContext): ScoreBreakdownItem {
  const current = ctx.refs.hrv28dAvg;
  const previous = ctx.refs.hrvPrev28dAvg;
  const pct = current != null && previous != null && previous !== 0
    ? ((current - previous) / previous) * 100

  if (pct == null) score = 0;
  else if (pct > 8) score = 18;
  else if (pct >= 3) score = 15;
  else if (pct >= -2) score = 11;
  else if (pct >= -7) score = 6;
  else score = 2;

  return {
    key: "hrv_28d_trend",
    label: "HRV 28d Trend",
    score,

export function classifyOscillator(composite: number): OscillatorClass {
  if (composite >= 85) return OscillatorClass.PEAK;
  if (composite >= 70) return OscillatorClass.STRONG_BUILD;
  if (composite >= 55) return OscillatorClass.CONTROLLED_BUILD;
  if (composite >= 40) return OscillatorClass.RESET;
  return OscillatorClass.RESENSITIZE;
}

export function calculateCompositeScores(
  acute: AcuteScoreResult,
  resource: ResourceScoreResult,
  seasonal: SeasonalScoreResult
): OscillatorScoreResult {
  const compositeScore = round(
    acute.score * 0.5 + resource.score * 0.3 + seasonal.score * 0.2,
    0
  );

  return {
    acuteScore: acute.score,
    resourceScore: resource.score,
    seasonalScore: seasonal.score,
    compositeScore,
    oscillatorClass: classifyOscillator(compositeScore),
  };
}

export function getCycleDay28(startDateIso: string, currentDateIso: string): number {
  const start = new Date(startDateIso + "T00:00:00");
  const current = new Date(currentDateIso + "T00:00:00");
  const msPerDay = 24 * 60 * 60 * 1000;
  const diffDays = Math.floor((current.getTime() - start.getTime()) / msPerDay);
  return (diffDays % 28 + 28) % 28 + 1;
}

export function getCycleWeekType(cycleDay28: number): CycleWeekType {
  if (cycleDay28 >= 1 && cycleDay28 <= 7) return CycleWeekType.PRIME;
  if (cycleDay28 >= 8 && cycleDay28 <= 14) return CycleWeekType.OVERLOAD;
  if (cycleDay28 >= 15 && cycleDay28 <= 21) return CycleWeekType.PEAK;
  return CycleWeekType.RESENSITIZE;
}

12. Daily decision engine

export interface RecommendationEngineContext {
  today: DailyVitalsRecord;
  refs: RollingReferences;
  baselines: UserVitalsBaselines;
  acute: AcuteScoreResult;
  resource: ResourceScoreResult;
  seasonal: SeasonalScoreResult;
  composite: OscillatorScoreResult;
  cycleDay28: number;
  cycleWeekType: CycleWeekType;
}

export function deriveFlags(ctx: RecommendationEngineContext): DailyRecommendationFlags {
  const hrvToday = ctx.today.heart.hrvMs;
  const hrv7 = ctx.refs.hrv7dAvg;
  const rhrToday = ctx.today.heart.restingHrBpm;
  const rhr7 = ctx.refs.rhr7dAvg;
  const sleep = ctx.today.sleep.sleepDurationMin;

  const suppressedHrv = hrvToday != null && hrv7 != null ? hrvToday < hrv7 * 0.9 : false;
  const elevatedRhr = rhrToday != null && rhr7 != null ? rhrToday > rhr7 + 4 : false;
  const lowSleep = sleep != null ? sleep < 360 : false;
  const cardioMonotony =
    Math.max(ctx.refs.cardioZone2Count7d, ctx.refs.cardioZone3Count7d, ctx.refs.cardioRecoveryCount7d) >= 5;
  const monthlyResensitizeOverride = ctx.cycleDay28 >= 22;
  const hardStopFatigue =
    suppressedHrv || elevatedRhr || lowSleep || ctx.composite.compositeScore < 55;

  return {
    hardStopFatigue,
    lowSleep,
    elevatedRhr,
    suppressedHrv,
    cardioMonotony,
    monthlyResensitizeOverride,
  };
}

export function recommendCardioMode(ctx: RecommendationEngineContext, flags: DailyRecommendationFlags): CardioMode {
  if (flags.hardStopFatigue) {
    if (ctx.composite.compositeScore < 40) return CardioMode.RECOVERY_WALK;
    return CardioMode.ZONE_2;
  }

  if (flags.monthlyResensitizeOverride && ctx.refs.cardioZone3Count7d >= 1) {
    return CardioMode.ZONE_2;
  }

  if (ctx.refs.cardioZone3Count7d >= 3) return CardioMode.ZONE_2;

  if (ctx.composite.compositeScore >= 85) return CardioMode.ZONE_3;
  if (ctx.composite.compositeScore >= 70) return CardioMode.ZONE_3;
  if (ctx.composite.compositeScore >= 55) return CardioMode.ZONE_2;
  if (ctx.composite.compositeScore >= 40) return CardioMode.ZONE_2;
  return CardioMode.RECOVERY_WALK;
}

export function recommendLiftMode(ctx: RecommendationEngineContext, flags: DailyRecommendationFlags): LiftMode {
  if (flags.hardStopFatigue) {
    return ctx.composite.compositeScore < 40 ? LiftMode.OFF : LiftMode.RECOVERY_PATTERNING;
  }

  switch (ctx.composite.oscillatorClass) {
    case OscillatorClass.PEAK:
      return LiftMode.NEURAL_TENSION;
    case OscillatorClass.STRONG_BUILD:
      return LiftMode.HYPERTROPHY_BUILD;
    case OscillatorClass.CONTROLLED_BUILD:
      return LiftMode.PUMP;
    case OscillatorClass.RESET:
      return LiftMode.RECOVERY_PATTERNING;
    case OscillatorClass.RESENSITIZE:
    default:
      return LiftMode.MOBILITY;
  }
}

export function recommendMacroDayType(
  ctx: RecommendationEngineContext,
  flags: DailyRecommendationFlags
): MacroDayType {
  if (flags.hardStopFatigue) {
    return ctx.composite.compositeScore < 40 ? MacroDayType.RESENSITIZE : MacroDayType.RESET;
  }

  if (flags.monthlyResensitizeOverride) {
    if (ctx.composite.compositeScore >= 85 && !flags.suppressedHrv && !flags.elevatedRhr) {
      return MacroDayType.BUILD;
    }
    return MacroDayType.RESET;
  }

  switch (ctx.composite.oscillatorClass) {
    case OscillatorClass.PEAK:
      return MacroDayType.SURGE;
    case OscillatorClass.STRONG_BUILD:
      return MacroDayType.BUILD;
    case OscillatorClass.CONTROLLED_BUILD:
      return flags.elevatedRhr ? MacroDayType.RESET : MacroDayType.BUILD;
    case OscillatorClass.RESET:
      return MacroDayType.RESET;
    case OscillatorClass.RESENSITIZE:
    default:
      return MacroDayType.RESENSITIZE;
  }
}

export function buildReasoning(
  ctx: RecommendationEngineContext,
  flags: DailyRecommendationFlags,
  cardio: CardioMode,
  lift: LiftMode,
  macro: MacroDayType
): string[] {
  const reasons: string[] = [];

  reasons.push(`Composite score ${ctx.composite.compositeScore} (${ctx.composite.oscillatorClass}).`);
  reasons.push(`Acute ${ctx.acute.score}, Resource ${ctx.resource.score}, Seasonal ${ctx.seasonal.score}.`);

  if (flags.suppressedHrv) reasons.push("HRV is suppressed versus 7-day baseline.");
  if (flags.elevatedRhr) reasons.push("RHR is elevated versus 7-day baseline.");
  if (flags.lowSleep) reasons.push("Sleep was below 6 hours.");
  if (flags.monthlyResensitizeOverride) reasons.push("Monthly cycle is in resensitize week (days 22-28).");
  if (flags.cardioMonotony) reasons.push("Recent cardio distribution is too repetitive.");

  reasons.push(`Assigned cardio mode: ${cardio}.`);
  reasons.push(`Assigned lift mode: ${lift}.`);
  reasons.push(`Assigned macro day: ${macro}.`);

  return reasons;
}

export function generateDailyRecommendation(ctx: RecommendationEngineContext): DailyRecommendation {
  const flags = deriveFlags(ctx);
  const recommendedCardioMode = recommendCardioMode(ctx, flags);
  const recommendedLiftMode = recommendLiftMode(ctx, flags);
  const recommendedMacroDayType = recommendMacroDayType(ctx, flags);
  const macroTargets = MACRO_TEMPLATES[recommendedMacroDayType];
  const mealTimingTargets = MEAL_TIMING_TEMPLATES[recommendedMacroDayType];
  const reasoning = buildReasoning(ctx, flags, recommendedCardioMode, recommendedLiftMode, recommendedMacroDayType);

  return {
    date: ctx.today.date,
    cycleDay28: ctx.cycleDay28,
    cycleWeekType: ctx.cycleWeekType,
    scores: ctx.composite,
    flags,
    recommendedCardioMode,
    recommendedLiftMode,
    recommendedMacroDayType,
    macroTargets,
    mealTimingTargets,
    reasoning,
  };
}

17. Practical implementation notes

The most important engineering choices:

Use smoothed trends

Do not use raw single-day smart-scale body fat or FFM to drive big decisions.

Use:
	•	7-day averages
	•	14-day linear slopes
	•	28-day rolling means

Confidence-weight noisy scale data

If body comp confidence is low, down-weight:
	•	BF%
	•	fat mass
	•	FFM

and rely more on:
	•	weight trend
	•	waist trend
	•	lift trend

Do not let one bad night nuke the system

Use hard-stop rules only when:
	•	multiple red flags
	•	or one severe flag

Keep recommendation deterministic

Same input should always produce same output.

That will matter a lot when you debug app behavior.
